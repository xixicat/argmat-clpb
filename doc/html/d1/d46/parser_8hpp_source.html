113 bool parser::Aspartix2DungAF(const std::string& fileName, DungAF& daf)
114 {
115  bool _is_successful = true;
116 
117  // The argument and attack regex expressions of Aspartix format
118  // arg(a1).
119  // att(a1, a2).
120  const boost::regex rgx_arg("arg\\(\\s*([a-zA-Z0-9_]+)\\s*\\)");
121  const boost::regex rgx_att("att\\(\\s*([a-zA-Z0-9_]+)\\s*,\\s*([a-zA-Z0-9_]+)\\s*\\)");
122 
123  // Before parsing, we need to clear the daf.
124  daf.clear();
125 
126  std::ifstream _infile;
127  _infile.open(fileName, ios_base::in);
128 
129  //check the file
130  if(!_infile)
131  {
132  cerr << "Can not open the file: " << fileName << endl;
133  exit(1);
134  }
135 
136  std::string _line_str;
137  size_type _line_num = 0;
138  boost::cmatch _result;
139 
140  while ( getline(_infile, _line_str) )
141  {
142  //_infile.getline()
143  //_infile >> _line_str;
144  ++_line_num;
145 
146  if(_line_str.length()<=4)
147  continue;
148 
149  if (boost::regex_search(_line_str.c_str(), _result, rgx_att)) // Find Attack
150  {
151  daf.addAttack(_result[1], _result[2]);
152  //cout << "Adding Attack: (" << _result[1] << "," << _result[2] << ")" << endl;
153  }
154  else if (boost::regex_search(_line_str.c_str(), _result, rgx_arg)) // Find Argument
155  {
156  daf.addArgument( _result[1] );
157  //cout << "Adding Argument: " << _result[1] << endl;
158  }
159  else
160  {
161  std::cerr << "Ignorance: " << fileName << endl
162  << " Line number(" << _line_num << "): " << _line_str << endl;
163  _is_successful = false;
164  }
165  }
166 
167  _infile.close();
168 
169  return _is_successful;
170 } // Aspartix2DungAF(const std::string& fileName, DungAF& daf)
