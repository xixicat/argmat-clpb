286 void ClpbProblem::parseAddtionalParams(const string& oargs)
287 {
288  m_otherArgs.clear();
289  if (oargs.length()==0){
290  return;
291  }
292  string sptor = ","; //The separator
293  size_type bgn=0, nd=0;
294  string arg;
295  for (nd = oargs.find_first_of(sptor);
296  nd != string::npos;
297  nd = oargs.find_first_of(sptor, bgn))
298  {
299  if (nd-bgn==0) { // avoid the case ,, and the , at the beginning.
300  cerr << "Wrong additional argument : " << oargs << endl;
301  cerr << "The right input for additional argument is: " << endl
302  << " arg1[,arg2,arg3,...]" << endl;
303  exit(1);
304  }
305 
306  arg = oargs.substr(bgn, nd - bgn);
307 
308  m_otherArgs.insert(arg);
309  bgn = nd + 1;
310  }
311 
312  // insert the last element
313  arg = oargs.substr(bgn);
314  if (arg.length()>0)
315  m_otherArgs.insert(arg);
316  else {
317  cerr << "Wrong additional argument : " << oargs << endl;
318  cerr << "The right input for additional argument is: " << endl
319  << " arg1[,arg2,arg3,...]" << endl;
