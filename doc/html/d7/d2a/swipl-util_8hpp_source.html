210  PlTimeQry(const char *module, const char *name, const PlTermv &av) {
211  atom_t mdl_atom = PL_new_atom(module);
212  atom_t time_atom = PL_new_atom("time");
213  module_t mdl = PL_new_module(mdl_atom);
214 
215  // the Goal
216  PlTermv cp_term(PlCompound(name, av));
217 
218  predicate_t p = PL_pred(PL_new_functor(time_atom, cp_term.size), mdl);
219 
220  PL_unregister_atom(mdl_atom);
221  PL_unregister_atom(time_atom);
222 
223  //PL_Q_NORMAL
224  //Normal operation. The debugger inherits its settings from the environment. If an excep-
225  //tion occurs that is not handled in Prolog, a message is printed and the tracer is started to
226  //debug the error.4
227  //PL_Q_NODEBUG
228  //Switch off the debugger while executing the goal. This option is used by many calls
229  //to hook-predicates to avoid tracing the hooks.
230  //An example is print/1 calling
231  //portray/1 from foreign code.
232  //PL_Q_CATCH EXCEPTION
233  //If an exception is raised while executing the goal, do not report it, but make it available
234  //for PL exception().
235  //PL_Q_PASS EXCEPTION
236  //As PL_Q_CATCH EXCEPTION, but do not invalidate the exception-term while calling
237  //PL_close_query(). This option is experimental.
238  qid = PL_open_query(mdl, PL_Q_CATCH_EXCEPTION, p, cp_term.a0);
239  if ( !qid )
240  throw PlResourceError();
241  }
242 
243  ~PlTimeQry() {
244  PL_cut_query(qid);
245  }
