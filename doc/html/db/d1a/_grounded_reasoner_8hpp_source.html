130 void GroundedReasoner::task_EC(const std::set<string>& argset)
131 {
132  if(argset.empty()) {
133  cerr << "The task DC needs additional argument (see option -a)" << endl;
134  exit(1);
135  }
136 
137  // Get the bitvector
138  bitvector bv_argset = m_daf.labelSet2bv(argset);
139 
140  bitvector gr_ext = getGroundedExtension();
141 
142  // The task EC is to enumerate all extensions that contain the *argset*.
143  // The grounded extension is unique. Therefore, if *argset* is included
144  // in the grounded extension, then we output the grounded extension;
145  // otherwise, output the emptyset.
146 
147  m_output << LEFT_LIMITER; // [
148  if (bv_argset.is_subset_of(gr_ext)){
149  printLabSet( gr_ext );
150  }
151  m_output << RIGHT_LIMITER; // ]
152 }
