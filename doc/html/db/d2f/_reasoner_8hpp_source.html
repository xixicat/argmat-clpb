484 argumatrix::bitvector Reasoner::getGroundedExtension()
485 {
486  bitvector _bv = bitvector::EmptySet(m_argNum);
487  bitvector _bv_next = characteristic(_bv);
488 
489  while (_bv != _bv_next)
490  {
491  _bv = _bv_next;
492  _bv_next = characteristic(_bv);
493  }
494 
495  return _bv_next;

499 vector<int> Reasoner::getGroundedIntVector()
500 {
501  // create m_argNum length vector with initial value 2 -- Unknown
502  vector<int> vecii(m_argNum, 2);
503 
504  bitvector gr_ext = bitvector::EmptySet(m_argNum);
505  bitvector gr_out;
506  bitvector _bv_last;
507  do {
508  _bv_last = gr_ext;
509  //gr_out = (m_BmAtkMtx * _bv_last); // S = R^+(X)
510  //gr_ext = ~(m_BmAtkMtx * (~gr_out)); // G = ~R^+(~S)
511  gr_out = getAttacked( _bv_last ); // S = R^+(X)
512  gr_ext = neutrality( ~gr_out ); // G = ~R^+(~S)
513  } while (gr_ext != _bv_last);
514 
515  // cout << "gr_ext" << gr_ext << endl;
516 
517  for(size_type i=0; i<m_argNum; ++i)
518  {
519  if( gr_ext[i] ) { vecii[i] = 1; }
520  else if ( gr_out[i] ) { vecii[i] = 0; }
521  }
522 
523  return vecii;
524 }

552 __inline
553 bool Reasoner::is_admissible(const bitvector& _bv)
554 {
555  // $S$ is an admissible extension iff $S \subseteq F(S) \cap N(S)$.
556  bitvector neu_s = neutrality(_bv);
557  bitvector f_s = neutrality( neu_s );
558  f_s &= neu_s;
559 
560  return _bv.is_subset_of(f_s);
561 }
