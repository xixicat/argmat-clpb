303 void PlReasoner::addBlTermToBvExts(const PlTerm& plt)
304 {
305  if ( !PL_is_list(plt.ref) )
306  {
307  throw PlTypeError("list", plt.ref);
308  return;
309  }
310 
311  //the list ref will get modified so copy it
312  term_t tail = PL_copy_term_ref(plt.ref);
313  term_t head = PL_new_term_ref();
314  int x, index = 0;
315  bitvector _bv(m_argNum, 0); // all bits are 0s.
316  while(PL_get_list(tail, head, tail)) {
317  PL_get_integer(head, &x);
318  if(x == 1) {
319  _bv.set(index);
320  }
321 
322  ++index;
323  }
324 
325  // The length of the list must be equal to the number of arguments.
326  assert(m_argNum == index);
327 
328  m_extensions.insert( _bv );
329 }
